public bool CommentsInsert(Comments objComments)
{

    string CustNameID = objComments.CustNameID;
    int Rating = objComments.Rating;
    string Comment = objComments.Comment;

    string connString = ConfigurationSettings.AppSettings.Get("connString");

    using (OleDbConnection cn = new OleDbConnection(connString))
    {
        using (OleDbCommand command = new OleDbCommand("INSERT INTO [CustomerComments] ([CustNameID], [Rating], [Comment]) VALUES (@CustNameID, @Rating, @Comment)", cn))
        {

            command.Parameters.Add("@CustNameID", OleDbType.VarWChar, 255).Direction = ParameterDirection.Input;
            command.Parameters.Add("@Rating", OleDbType.Integer, 4).Direction = ParameterDirection.Input;
            command.Parameters.Add("@Comment", OleDbType.VarWChar, 255).Direction = ParameterDirection.Input;

            command.Parameters["@CustNameID"].Value = CustNameID;
            command.Parameters["@Rating"].Value = Rating;
            command.Parameters["@Comment"].Value = Comment;

            cn.Open();
            command.ExecuteNonQuery();

        }

        cn.Close();

    }

    return true;

}
